
main()

	get semaphores
	initiate system v message queues

	create 3 named tables (london, lincoln, salisbury)

	create 7 mock users (phill, allan, russ, will, tim)

	start the parent socket accepting loop: listen_socket()

listen_socket()

	wait for connections, spawn child_conn() for each socket

child_conn()

	accept socket
	prompt for username
	prompt for password (and check equal to password)
	prompt for table number
	create system v message queue, put socket, username on queue
	quit thread

table

	wait for a message queue sent to the table
	message queue contains player.
	
	wait until 6 players join, call game()

game()

	application logic (deal cards, round of betting etc)
